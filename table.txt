create table admin(
    	admin_ID VARCHAR2(20 BYTE), 
	admin_PW VARCHAR2(200 BYTE), 
	admin_NAME VARCHAR2(40 BYTE), 
	admin_PH1 VARCHAR2(13 BYTE), 
	admin_PH2 VARCHAR2(13 BYTE),

create table goods(
    book_isbn varchar2(40),
    book_name varchar2(200),
    book_author_name varchar2(20),
    book_category varchar2(20),
    partner_name varchar2(50),
    book_date timestamp(6),
    book_price varchar2(200),
    book_page_num varchar2(20),
    book_length varchar2(20),
    book_sub varchar2(200),
    book_image varchar2(300),
    book_intro varchar2(200),
    book_author_intro varchar2(200),
    book_list varchar2(200),
    book_count varchar2(100),
    book_regist timestamp(6) default sysdate,
    partner_num varchar2(200),
    book_num varchar2(2000)
);

페이지수, 	book_page_num
